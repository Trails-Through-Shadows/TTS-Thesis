{% base it on the "list" style
  \setglossarystyle{list}% 

  \renewcommand*{\glossentry}[2]{% 
    % name
    \item[\glsentryitem{##1}\glstarget{##1}{\glossentryname{##1}}] -- % 
    % publisher
    \glossentrydesc{##1}%
    % release year (if present)
    \ifthenelse{\equal{\glsentryuseri{##1}}{}}{}{, \glsentryuseri{##1}}%
    % page numbers
    \space(##2)\\%
    % boardgamegeek link
    BoardGameGeek: \url{\glsentryuserii{##1}}%
    % publisher link (if present)
    \ifthenelse{\equal{\glsentryuseriii{##1}}{}}{}{\\Vydavatel: \url{\glsentryuseriii{##1}}}%
  }%
}
\setglossarystyle{gamelist}
